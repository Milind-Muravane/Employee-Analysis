import pandas as pd
import matplotlib.pyplot as plt
import random

departments = (
    ["Marketing"] * 15 +
    ["Sales"] * 20 +
    ["HR"] * 20 +
    ["Finance"] * 20 +
    ["IT"] * 25
)

regions = ["North", "South", "East", "West"]

df = pd.DataFrame({
    "EmployeeID": range(1, 101),
    "Department": departments,
    "Region": [random.choice(regions) for _ in range(100)],
    "PerformanceScore": [random.randint(70, 100) for _ in range(100)]
})

marketing_count = (df["Department"] == "Marketing").sum()
print("Frequency count for Marketing department:", marketing_count)

plt.figure(figsize=(8, 5))
plt.hist(df["Department"], color="skyblue", edgecolor="black")
plt.title("Department Distribution Histogram")
plt.xlabel("Department")
plt.ylabel("Frequency")
plt.xticks(rotation=45)
plt.tight_layout()
plt.savefig("department_histogram.png")
plt.show()
